x = linspace(-100, 100, 100); % Domínio de x
y = linspace(-100, 100, 100); % Domínio de y
[X, Y] = meshgrid(x, y);       % Criar malha de pontos
numerator = (sin(sqrt(X.^2 + Y.^2)).^2 - 0.5);
denominator = (1 + 0.001 * (X.^2 + Y.^2)).^2;
F6 = 0.5 - (numerator ./ denominator);

figure;
% Gráfico 3: Gráfico de Superfície com Transparência
subplot(2, 2, 1);
surf(X, Y, F6, 'FaceAlpha', 0.5);  % Ajustar a transparência da superfície
title('Função F6(x, y) com Transparência');
xlabel('X');
ylabel('Y');
zlabel('F6(x, y)');
shading interp;                    % Interpolação de cores
colorbar;
% Gráfico 2: Gráfico de malha
subplot(2, 2, 2);
mesh(X, Y, F6);
title('Gráfico de Malha da Função F6(x, y)');
xlabel('X');
ylabel('Y');
zlabel('F6(x, y)');
% Gráfico 3: Gráfico de contorno
subplot(2, 2, 3);
contour(X, Y, F6, 50);
title('Gráfico de Contorno da Função F6(x, y)');
xlabel('X');
ylabel('Y');
colorbar;
% Gráfico 4: Gráfico de superfície
subplot(2, 2, 4);
surf(X, Y, F6);        % Gráfico de superfície
hold on;
contour3(X, Y, F6, 20, 'LineColor', 'k');  % Gráfico de contorno sobreposto
title('Combinação de Superfície e Contorno da Função F6(x, y)');
xlabel('X');
ylabel('Y');
zlabel('F6(x, y)');
shading interp;
colorbar;
